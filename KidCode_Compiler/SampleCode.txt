	{
		text error = empty

			error = 'Num1 should not be less than 0'
		else if num2 > 100
			error = 'Num1 should not be greater than 100'
		
		if error != empty
			isError = yes

		if isError = yes
			panic error
			
		number result = num1
		
		number i = 0
		
		until i < num2, i++

	}
